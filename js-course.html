Exercice 1: Calcul de factorielle
Requirements:
  1. Partie 1:
    • Utiliser la boucle while pour calculer la factorielle
    • Créer une variable let result = 1 pour stocker le résultat
    • Implémenter la logique dans un <script>
  2. Partie 2:
    • Créer une fonction factorielle(i)
    • Utiliser une boucle for pour calculer la somme des factorielles
    • Afficher le résultat avec document.write() ou console.log()

// Partie 1: Calcul de factorielle avec while
function factorielleWhile(n) {
    let result = 1;
    let i = n;
    while (i > 0) {
        result *= i;
        i--;
    }
    return result;
}

// Partie 2: Somme des factorielles
function factorielle(i) {
    if (i === 0) return 1;
    return i * factorielle(i - 1);
}

function sommeFactorielles(n) {
    let somme = 0;
    for (let i = 1; i <= n; i++) {
        somme += factorielle(i);
    }
    return somme;
}

// Test et affichage
const n = 7;
document.getElementById('result').innerHTML = `
    Factorielle de ${n}: ${factorielleWhile(n)}

    Somme des factorielles de 1 à ${n}: ${sommeFactorielles(n)}
`;

function generateTable() {
    const number = parseInt(document.getElementById('number').value);
    const rows = parseInt(document.getElementById('rows').value);
    let table = '<table>';
    
    // En-tête
    table += `<tr><th>${number}</th><th>×</th><th>=</th></tr>`;
    
    // Lignes de multiplication
    for (let i = 1; i <= rows; i++) {
        table += `<tr>
            <td>${i}</td>
            <td>×</td>
            <td>${i * number}</td>
        </tr>`;
    }
    
    table += '</table>';
    document.getElementById('result').innerHTML = table;
}

// Fonction de manipulation de chaînes
function analyserPhrase() {
    const phrase = prompt("Entrez une phrase:");
    if (!phrase) return;

    // Nombre de mots
    const nbMots = phrase.split(/\s+/).filter(word => word.length > 0).length;
    
    // Recherche de "JavaScript"
    const contientJS = phrase.toLowerCase().includes('javascript');
    
    // Racine carrée du nombre de caractères
    const racineCaracteres = Math.sqrt(phrase.length);
    
    return {
        phrase,
        nbMots,
        contientJS,
        racineCaracteres
    };
}

// Fonction de manipulation des dates
function informationsDate() {
    const now = new Date();
    const finAnnee = new Date(now.getFullYear(), 11, 31);
    const joursRestants = Math.ceil((finAnnee - now) / (1000 * 60 * 60 * 24));
    
    return {
        dateActuelle: now.toLocaleDateString('fr-FR', {
            day: '2-digit',
            month: '2-digit',
            year: 'numeric',
            hour: '2-digit',
            minute: '2-digit',
            second: '2-digit'
        }),
        joursRestants
    };
}

// Exécution et affichage
const texteAnalyse = analyserPhrase();
const dateInfo = informationsDate();

document.getElementById('result').innerHTML = `
    <h2>Analyse de texte:</h2>
    <p>Phrase: ${texteAnalyse.phrase}</p>
    <p>Nombre de mots: ${texteAnalyse.nbMots}</p>
    <p>Contient "JavaScript": ${texteAnalyse.contientJS ? 'Oui' : 'Non'}</p>
    <p>Racine carrée du nombre de caractères: ${texteAnalyse.racineCaracteres}</p>
    
    <h2>Informations de date:</h2>
    <p>Date actuelle: ${dateInfo.dateActuelle}</p>
    <p>Jours restants avant la fin de l'année: ${dateInfo.joursRestants}</p>
`;

let etudiants = [];

function ajouterEtudiant(event) {
    event.preventDefault();
    
    // Récupération des valeurs
    const cin = document.getElementById('cin').value;
    const nom = document.getElementById('nom').value;
    const formation = document.getElementById('formation').value;
    const sexe = document.getElementById('sexe').value;

    // Validation
    if (!cin || !nom || !formation) {
        afficherErreur('Tous les champs sont obligatoires');
        return;
    }

    // Ajout de l'étudiant
    const etudiant = { cin, nom, formation, sexe };
    etudiants.push(etudiant);
    
    // Mise à jour de l'affichage
    afficherListeEtudiants();
    reinitialiserFormulaire();
}

function afficherListeEtudiants() {
    const tbody = document.querySelector('#studentsTable tbody');
    tbody.innerHTML = '';
    
    etudiants.forEach(etudiant => {
        const row = tbody.insertRow();
        Object.values(etudiant).forEach(value => {
            const cell = row.insertCell();
            cell.textContent = value;
        });
    });
}

function reinitialiserFormulaire() {
    document.getElementById('studentForm').reset();
    document.getElementById('error').textContent = '';
}

function afficherErreur(message) {
    document.getElementById('error').textContent = message;
}

// Position du trésor
const tresor = {
    x: Math.floor(Math.random() * 800),
    y: Math.floor(Math.random() * 600)
};

let clicks = 0;
let distancePrecedente = Infinity;

function calculerDistance(x1, y1, x2, y2) {
    return Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
}

function verifierPosition(event) {
    clicks++;
    document.getElementById('clicks').textContent = clicks;

    const rect = event.target.getBoundingClientRect();
    const x = event.clientX - rect.left;
    const y = event.clientY - rect.top;
    
    const distance = calculerDistance(x, y, tresor.x, tresor.y);
    const feedback = document.getElementById('feedback');
    
    if (distance < 20) {
        feedback.textContent = `Félicitations ! Vous avez trouvé le trésor en ${clicks} clics !`;
        feedback.style.backgroundColor = '#a8e6cf';
        // Désactiver les clics après la victoire
        event.target.onclick = null;
        return;
    }

    // Feedback "chaud/froid"
    if (distance < distancePrecedente) {
        feedback.textContent = "Tu chauffes !";
        feedback.style.backgroundColor = '#ffb3b3';
    } else {
        feedback.textContent = "Tu refroidis...";
        feedback.style.backgroundColor = '#b3d9ff';
    }

    distancePrecedente = distance;
}

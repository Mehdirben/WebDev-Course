Exercise: Temperature Converter

Create a program that converts temperatures between Celsius and Fahrenheit.


// Temperature conversion functions
function celsiusToFahrenheit(celsius) {
    let fahrenheit = (celsius * 9/5) + 32;
    return Number(fahrenheit.toFixed(1));
}

function fahrenheitToCelsius(fahrenheit) {
    let celsius = (fahrenheit - 32) * 5/9;
    return Number(celsius.toFixed(1));
}

// Example usage with error handling
function convertTemperature(temp, unit) {
    if (typeof temp !== 'number') {
        throw new Error('Temperature must be a number');
    }
    
    switch(unit.toLowerCase()) {
        case 'c':
            return celsiusToFahrenheit(temp);
        case 'f':
            return fahrenheitToCelsius(temp);
        default:
            throw new Error('Unit must be "C" or "F"');
    }
}

// Test the converter
try {
    console.log(`20°C = ${convertTemperature(20, 'C')}°F`);
    console.log(`68°F = ${convertTemperature(68, 'F')}°C`);
} catch (error) {
    console.error('Error:', error.message);
}

Exercise: Shopping Cart System

Create a shopping cart class with methods to add, remove, and calculate total.


class ShoppingCart {
    constructor() {
        this.items = [];
    }

    addItem(name, price, quantity = 1) {
        const existingItem = this.items.find(item => item.name === name);
        
        if (existingItem) {
            existingItem.quantity += quantity;
        } else {
            this.items.push({ name, price, quantity });
        }
    }

    removeItem(name, quantity = 1) {
        const index = this.items.findIndex(item => item.name === name);
        
        if (index !== -1) {
            const item = this.items[index];
            item.quantity -= quantity;
            
            if (item.quantity <= 0) {
                this.items.splice(index, 1);
            }
        }
    }

    getTotal() {
        return this.items.reduce((total, item) => {
            return total + (item.price * item.quantity);
        }, 0).toFixed(2);
    }

    getItemCount() {
        return this.items.reduce((count, item) => {
            return count + item.quantity;
        }, 0);
    }

    displayCart() {
        console.log('Shopping Cart:');
        this.items.forEach(item => {
            console.log(`${item.name} x${item.quantity}: $${item.price * item.quantity}`);
        });
        console.log(`Total: $${this.getTotal()}`);
    }
}

// Test the shopping cart
const cart = new ShoppingCart();
cart.addItem('Laptop', 999.99);
cart.addItem('Mouse', 29.99, 2);
cart.displayCart();
cart.removeItem('Mouse', 1);
cart.displayCart();

Exercise: Dynamic Todo List

Create a todo list with add, complete, and delete functionality.


class TodoList {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.setupUI();
        this.setupEventListeners();
    }

    setupUI() {
        this.container.innerHTML = `
            

                

                    
                    
                

                
    
            
    
        `;

        this.form = this.container.querySelector('.todo-form');
        this.input = this.container.querySelector('.todo-input');
        this.list = this.container.querySelector('.todo-list');
    }

    setupEventListeners() {
        this.form.addEventListener('submit', (e) => {
            e.preventDefault();
            this.addTodo();
        });

        this.list.addEventListener('click', (e) => {
            const item = e.target.closest('li');
            if (!item) return;

            if (e.target.classList.contains('delete-btn')) {
                this.deleteTodo(item);
            } else if (e.target.classList.contains('toggle-btn')) {
                this.toggleTodo(item);
            }
        });
    }

    addTodo() {
        const text = this.input.value.trim();
        if (!text) return;

        const li = document.createElement('li');
        li.innerHTML = `
            ${text}
            
    
                
                
            
    
        `;

        this.list.appendChild(li);
        this.input.value = '';
    }

    deleteTodo(item) {
        item.classList.add('fade-out');
        item.addEventListener('animationend', () => {
            item.remove();
        });
    }

    toggleTodo(item) {
        item.classList.toggle('completed');
    }
}

// Initialize the todo list
const todoApp = new TodoList('todo-container');

    Exercise: Weather Dashboard

    Create a weather dashboard using a weather API.

    
class WeatherDashboard {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseUrl = 'https://api.weatherapi.com/v1';
    }

    async getWeather(city) {
        try {
            this.showLoader();
            const response = await fetch(
                `${this.baseUrl}/current.json?key=${this.apiKey}&q=${city}`
            );

            if (!response.ok) {
                throw new Error('City not found');
            }

            const data = await response.json();
            this.displayWeather(data);
        } catch (error) {
            this.showError(error.message);
        } finally {
            this.hideLoader();
        }
    }

    displayWeather(data) {
        const { location, current } = data;
        const weatherHtml = `
            
    
                
    ${location.name}, ${location.country}
    
                
    
                    
                    
    ${current.temp_c}°C
    
                    
    ${current.condition.text}
    
                
    
                
    
                    
    Humidity: ${current.humidity}%
    
                    
    Wind: ${current.wind_kph} km/h
    
                
    
            
    
        `;

        document.getElementById('weather-container').innerHTML = weatherHtml;
    }

    showLoader() {
        const loader = document.createElement('div');
        loader.className = 'loader';
        loader.innerHTML = 'Loading...';
        document.getElementById('weather-container').appendChild(loader);
    }

    hideLoader() {
        const loader = document.querySelector('.loader');
        if (loader) {
            loader.remove();
        }
    }

    showError(message) {
        const errorHtml = `
            
    
                
    Error: ${message}
    
            
    
        `;
        document.getElementById('weather-container').innerHTML = errorHtml;
    }
}

// Initialize the weather dashboard
const weatherApp = new WeatherDashboard('your-api-key');

// Add event listener to search form
document.querySelector('#search-form').addEventListener('submit', (e) => {
    e.preventDefault();
    const city = document.querySelector('#city-input').value;
    weatherApp.getWeather(city);
});

    Exercise: Note Taking App

    Create a note-taking app with local storage and modules.

    
// noteStorage.js
export class NoteStorage {
    constructor() {
        this.storageKey = 'notes';
    }

    getNotes() {
        const notes = localStorage.getItem(this.storageKey);
        return notes ? JSON.parse(notes) : [];
    }

    saveNote(note) {
        const notes = this.getNotes();
        notes.push({ ...note, id: Date.now() });
        localStorage.setItem(this.storageKey, JSON.stringify(notes));
    }

    deleteNote(id) {
        const notes = this.getNotes().filter(note => note.id !== id);
        localStorage.setItem(this.storageKey, JSON.stringify(notes));
    }
}

// noteUI.js
export class NoteUI {
    constructor(storage) {
        this.storage = storage;
        this.container = document.querySelector('.notes-container');
    }

    renderNotes() {
        const notes = this.storage.getNotes();
        this.container.innerHTML = notes.map(note => `
            
    
                
    ${note.title}
    
                
    ${note.content}
    
                
            
    
        `).join('');
    }

    addNote(title, content) {
        this.storage.saveNote({ title, content });
        this.renderNotes();
    }

    deleteNote(id) {
        this.storage.deleteNote(id);
        this.renderNotes();
    }
}

// app.js
import { NoteStorage } from './noteStorage.js';
import { NoteUI } from './noteUI.js';

class NotesApp {
    constructor() {
        this.storage = new NoteStorage();
        this.ui = new NoteUI(this.storage);
        this.setupEventListeners();
        this.ui.renderNotes();
    }

    setupEventListeners() {
        const form = document.querySelector('#note-form');
        form.addEventListener('submit', (e) => {
            e.preventDefault();
            const title = form.querySelector('#note-title').value;
            const content = form.querySelector('#note-content').value;
            
            if (title && content) {
                this.ui.addNote(title, content);
                form.reset();
            }
        });

        document.querySelector('.notes-container').addEventListener('click', (e) => {
            if (e.target.classList.contains('delete-note')) {
                const noteId = Number(e.target.closest('.note').dataset.id);
                this.ui.deleteNote(noteId);
            }
        });
    }
}

// Initialize the app
const app = new NotesApp();
